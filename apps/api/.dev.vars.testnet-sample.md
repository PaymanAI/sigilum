# `.dev.vars` Testnet Sample

###################
# Testnet profile that mirrors production behavior but runs synchronously (no blockchain queue required).
#
# Notes:
# - `BLOCKCHAIN_MODE=sync` submits chain writes inline and does not require `BLOCKCHAIN_QUEUE`.
# - Useful for end-to-end validation against testnet without queue setup.
# - `RESEND_API_KEY` is the preferred transport for final webhook failure alert emails.
# - Final webhook-failure alerts are sent to the namespace owner's email (`users.email`) when `payload.namespace` is present.
###################

ENVIRONMENT=staging
ADAPTER_PROVIDER=cloudflare

JWT_SECRET=replace-with-staging-secret
JWT_EXPIRY=7d
ALLOWED_ORIGINS=https://staging-dashboard.sigilum.dev

BLOCKCHAIN_MODE=sync
BLOCKCHAIN_NETWORK=testnet
BLOCKCHAIN_RPC_URL=https://sepolia.base.org
SIGILUM_REGISTRY_ADDRESS=0xREPLACE_TESTNET_REGISTRY
RELAYER_PRIVATE_KEY=0xREPLACE_TESTNET_RELAYER_KEY
MAX_PENDING_AUTHORIZATIONS=20
PENDING_AUTHORIZATION_EXPIRY_HOURS=24
CHALLENGE_EXPIRY_HOURS=1
DEBUG_LOGGING=true

WEBHOOK_RETRY_WINDOW_HOURS=24
WEBHOOK_FAILURE_THRESHOLD=10
WEBHOOK_ALERT_EMAIL_FROM=alerts@your-domain.com
RESEND_API_KEY=re_replace_testnet_key
