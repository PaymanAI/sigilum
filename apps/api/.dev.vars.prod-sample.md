# `.dev.vars` Production Sample

###################
# Production-like profile with blockchain writes queued.
#
# Notes:
# - Requires queue bindings from `wrangler.toml`:
# - `BLOCKCHAIN_QUEUE`
# - `WEBHOOK_QUEUE`
# - `BLOCKCHAIN_MODE=queue` is the intended durable mode for production.
# - `RESEND_API_KEY` is the preferred transport for final webhook failure alert emails.
# - Final webhook-failure alerts are sent to the namespace owner's email (`users.email`) when `payload.namespace` is present.
###################

ENVIRONMENT=production
ADAPTER_PROVIDER=cloudflare

JWT_SECRET=replace-with-strong-secret
JWT_EXPIRY=7d
ALLOWED_ORIGINS=https://dashboard.sigilum.dev

BLOCKCHAIN_MODE=queue
BLOCKCHAIN_NETWORK=mainnet
BLOCKCHAIN_RPC_URL=https://mainnet.base.org
SIGILUM_REGISTRY_ADDRESS=0xREPLACE_MAINNET_REGISTRY
RELAYER_PRIVATE_KEY=0xREPLACE_MAINNET_RELAYER_KEY
MAX_PENDING_AUTHORIZATIONS=20
PENDING_AUTHORIZATION_EXPIRY_HOURS=24
CHALLENGE_EXPIRY_HOURS=1
DEBUG_LOGGING=false

WEBHOOK_RETRY_WINDOW_HOURS=24
WEBHOOK_FAILURE_THRESHOLD=10
WEBHOOK_ALERT_EMAIL_FROM=alerts@your-domain.com
RESEND_API_KEY=re_replace_prod_key
